\chapter{List of Symbols}

\symb{$i$}{Unit imaginary number}\\
\symb{$j$, $ j' $}{Fine structure or total spin angular momentum quantum number of individual atoms as the ground state or excited state, respectively}\\
\symb{$f$, $ f' $}{Hyperfine structure or total atomic angular momentum quantum number of individual atoms as the ground state or excited state, respectively}\\
\symb{$J$, $ J' $}{The collective fine structure or total spin angular momentum quantum number of an atom ensemble as the ground state or excited state, respectively}\\
\symb{$F$, $ F' $}{The collective hyperfine structure or total atomic angular momentum quantum number of an atom ensemble as the ground state or excited state, respectively}\\
\symb{$\hat{F}$, $ \hat{f} $}{The collective and individual spin operators}\\
\symb{$ \hat{\mathbf{S}}$, $\hat{S}_i$}{Stokes operator and its $i$-th component $ (i=1,2,3) $}\\
\symb{$\mathbf{e}_i$}{The unit vector along the $ i $ direction}\\
\symb{$\delta_{jk}$}{Kronecker delta function}\\
\symb{$\delta(x)$}{Dirac delta function}\\
\symb{$\dt A$}{Time derivative of $A$}\\
\symb{$A^\dagger$}{Hermitian conjugate of $A$}\\
\symb{$A^*$}{Complex conjugate of $A$}\\
\symb{$A^\transp$}{Transpose of $A$}\\
\symb{$\det(A)$}{Determinant of $A$}\\
\symb{$\tr(A)$}{Trace of $A$}\\
\symb{$\av{A}$}{Expectation value of $A$}\\
\symb{$[A,B]$}{Commutator of $A$ and $B$}\\
\symb{$\{A,B\}$}{Anticommutator of $A$ and $B$}\\
%\symb{$\Colon\,\quad\,\Colon$}{Normal ordering of modal operators}\\
\symb{$\re(z)$}{Real part of $z$}\\
\symb{$\im(z)$}{Imaginary part of $z$}\\

\symb{$\identity$, $\nullmatrix$}{The identity matrix and null matrix}\\
%\symb{$\mathfrak{m}$}{Fraktur fonts for symplectic vectors}\\
\symb{$\br$,\,$\br'$}{Full spatial position vectors of a detector and the source}\\
\symb{$\br_\perp$,\,$\br'_\perp$}{The transverse spatial position vectors of a detector and the source}\\
%\symb{$\vec{x}$}{Vectors of other degrees of freedom}\\
\symb{$\tensor{G}$}{Tensors of spatial degrees of freedom}\\
\symb{$\tensor{G}(\br,\br')$}{The dyadic Green's function or Green's function tensor responded at position $ \br $ due to a source at $ \br' $}\\
\symb{$\hat{\tensor{\alpha}}$}{Polarizability tensor operators of an atom}\\
\symb{$\Gamma_0$}{The natural linewidth of an atom}\\
\symb{$\hat{T}^{(r)}$}{Rank-$r$ tensor of operators}\\
\symb{$C^{(r)}_{jfj'f'}$}{Rank-$r$ hyperfine structure atomic transition coefficients connecting ground states with fine structure quantum number $ j $ and hyperfine structure $ f $ and the excited state counterpart denoted with prime ($ \prime $) }\\
\symb{$\hat{a}$, $\hat{a}^\dagger$ ($\hat{b}$, $\hat{b}^\dagger$)}{Creation and annihilation operators for bosonic modes}\\
\symb{$\beta_0$}{Propagation constant of a guided mode}\\
\symb{$ k $, $k_0$}{The total wave vector number of a propagating light}\\
\symb{$ \omega $, $\omega_0$}{Angular frequency of a light or an electromagnetic wave}\\
\symb{$\lambda$}{Wavelength of a light or an electromagnetic wave}\\
\symb{$\ket{\vac}$}{Vacuum state}\\
\symb{$\mathrm{c.c.}$}{Complex conjugate}\\
\symb{$\mathrm{H.c.}$}{Hermitian conjugate}
